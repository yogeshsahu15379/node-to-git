# Spacing values

The consistent finite set of spacings used throughout components.

<Story name="Spacing">
  {() => {
    const spacings = Object.entries(tailwindThemeConfig.spacing).map(
      ([name, value]) => html`
        <tr>
          <td class="py-1 text-center border-b border-neutrals-gray3">
            ${name}
          </td>
          <td class="py-1 text-center border-b border-neutrals-gray3">
            ${value}
          </td>
          <td class="py-1 text-center border-b border-neutrals-gray3">
            ${extractPx(value)}
          </td>
          <td class="py-1 text-center border-b border-neutrals-gray3">
            <div class="h-4 bg-signature-blue1 w-${name}"></div>
          </td>
        </tr>
      `,
    )
    return html`
      <table class="w-full table-auto shadow-lg border border-neutrals-gray3">
        <thead class="text-xl font-display">
          <tr class="border border-neutrals-gray3 bg-neutrals-gray1">
            <th scope="col" class="p-3 font-normal">Name</th>
            <th scope="col" class="p-3 font-normal">Size</th>
            <th scope="col" class="p-3 font-normal">Pixels</th>
            <th scope="col" class="p-3 font-normal">Example</th>
          </tr>
        </thead>
        <tbody class="font-body">
          ${spacings}
        </tbody>
      </table>
    `
  }}
</Story>
